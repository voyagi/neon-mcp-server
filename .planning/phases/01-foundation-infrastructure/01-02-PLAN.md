// Customer status enum — errorMap shows invalid value + valid options
const CustomerStatus = z.enum(["active", "inactive", "lead"], {
  errorMap: (issue, ctx) => {
    if (issue.code === "invalid_enum_value") {
      return { message: `'${ctx.data}' is not valid. Expected: active, inactive, lead` };
    }
    return { message: ctx.defaultError };
  }
});

**3. `formatValidationError()` helper:**

```typescript
export function formatValidationError(error: z.ZodError): string {
  return error.errors
    .map(err => {
      const path = err.path.length > 0 ? `${err.path.join(".")}: ` : "";
      return `${path}${err.message}`;
    })
    .join("; ");
}
```

This collects ALL errors from `.safeParse()` into one string separated by semicolons. Claude sees every issue at once and can fix them in a single retry.
